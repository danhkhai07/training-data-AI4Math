    \textbf{Lời giải}

\noindent a)
$$
A = \begin{bmatrix} 1 & 3 & 5 & -1 \\ 2 & -1 & -1 & 4 \\ 5 & 1 & -1 & 7 \\ 7 & 7 & 9 & 1 \end{bmatrix} \xrightarrow{\substack{L_2-2L_1 \rightarrow L_2 \\ L_3-5L_1 \rightarrow L_3 \\ L_4-7L_1 \rightarrow L_4}}
\begin{bmatrix} 1 & 3 & 5 & -1 \\ 0 & -7 & -11 & 6 \\ 0 & -14 & -26 & 12 \\ 0 & -14 & -26 & 8 \end{bmatrix} \xrightarrow{\substack{L_3-2L_2 \rightarrow L_3 \\ L_4-2L_2 \rightarrow L_4}}
\begin{bmatrix} 1 & 3 & 5 & -1 \\ 0 & -7 & -11 & 6 \\ 0 & 0 & -4 & 0 \\ 0 & 0 & -4 & -4 \end{bmatrix}
$$

$$
\rightarrow \begin{bmatrix} 1 & 3 & 5 & -1 \\ 0 & -7 & -11 & 6 \\ 0 & 0 & -4 & 0 \\ 0 & 0 & 0 & -4 \end{bmatrix}. \quad \text{Vậy } r(A) = 4.
$$

\noindent \underline{\textbf{Cách 2:}} $\det A = -112 \neq 0, A$ là ma trận vuông cấp $4 \Rightarrow \text{rank } A = 4$.
