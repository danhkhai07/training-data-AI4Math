\subsection*{Lời giải}

$$

$$

\textbf{b)}
$$
B = \begin{vmatrix}
a+b & ab & a^2+b^2 \\
b+c & bc & b^2+c^2 \\
c+a & ca & c^2+a^2
\end{vmatrix}
= \begin{vmatrix}
a-c & b(a-c) & (a-c)(a+c) \\
b-a & c(b-a) & (b-a)(b+a) \\
c+a & ca & c^2+a^2
\end{vmatrix}
(L_1 - L_2 \to L_1; L_2 - L_3 \to L_2)
$$
$$
= (a-c)(b-a) \begin{vmatrix}
1 & b & a+c \\
1 & c & b+a \\
c+a & ca & c^2+a^2
\end{vmatrix}
= (a-c)(b-a) \begin{vmatrix}
1 & b & a+c \\
0 & c-b & b-c \\
c+a & ca & c^2+a^2
\end{vmatrix}
(L_2 - L_1 \to L_2)
$$
$$
= (a-c)(b-a) \begin{vmatrix}
1 & a+b+c & a+c \\
0 & 0 & b-c \\
c+a & a^2+c^2+ac & c^2+a^2
\end{vmatrix}
= (a-c)(b-a)(c-b) [a^2+c^2+ac - (a+c)(a+b+c)]
$$
$$
= (a-b)(b-c)(c-a)(ab+bc+ca).
$$
