\noindent \textbf{Bài 3.} Cho ma trận $A = \begin{bmatrix} 1 & -2 & 3 \\ 2 & -4 & 1 \\ 3 & -5 & 3 \end{bmatrix}$ và đa thức $f(x)=3x^2-2x+5$. Tính $f(A)$.

\begin{center}
    \textbf{Lời giải}
\end{center}

$$
A = \begin{bmatrix} 1 & -2 & 3 \\ 2 & -4 & 1 \\ 3 & -5 & 3 \end{bmatrix} \Rightarrow A^2 = \begin{bmatrix} 6 & -9 & 10 \\ -3 & -7 & 5 \\ 2 & -1 & 13 \end{bmatrix}
$$

$$
\Rightarrow f(A) = 3A^2 - 2A + 5E = \begin{bmatrix} 21 & -23 & 24 \\ -13 & 34 & 13 \\ 0 & 7 & 38 \end{bmatrix}
$$
