\section*{Problem}

\text{Determine the value(s) of $k$ for which the following system of linear equations has exactly}\
\text{one solution, and find that solution:}
$$
\begin{cases}
x - y + 2z = 0 \\
-x + y - z = 0 \\
x + ky + z = 0
\end{cases}
$$
