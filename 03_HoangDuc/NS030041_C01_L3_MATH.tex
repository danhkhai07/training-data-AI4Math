\section*{Problem}

\text{Let $\mathbf{A} = \begin{pmatrix}
2 & 3 & 5 \\
-5 & 1 & -4 \\
-3 & -1 & -4 \\
1 & 0 & 1
\end{pmatrix}$. Observe that the third column is the sum of the first two columns.} \\
\text{Find a non-trivial solution to $\mathbf{A}\mathbf{x} = \mathbf{0}$.}
