\section*{Problem}

\text{Find the solution to the system of linear equations with the following augmented matrices}\\
\text{using Gaussian elimination:}

\subsection*{(a)}
$$
\begin{pmatrix}
1 & -1 & 0 & 3 \\
0 & 1 & -2 & 1 \\
0 & 0 & 1 & -1
\end{pmatrix}
$$

\subsection*{(b)}
$$
\begin{pmatrix}
1 & 2 & 1 & 0 \\
0 & 0 & 1 & -1 \\
0 & 0 & 0 & 0
\end{pmatrix}
$$

\subsection*{(c)}
$$
\begin{pmatrix}
2 & 1 & -1 & 3 \\
1 & -1 & 1 & 0 \\
0 & 1 & 2 & 1
\end{pmatrix}
$$

\subsection*{(d)}
$$
\begin{pmatrix}
2 & 1 & 1 & 0 \\
1 & -2 & 1 & -2 \\
1 & 0 & 1 & 0
\end{pmatrix}
$$
