\section*{Problem}

\text{Let $T: \mathbf{R}^3 \to \mathbf{R}^3$ be a linear transformation defined by:}
$$
T \begin{pmatrix} 1 \\ 0 \\ 0 \end{pmatrix} = \begin{pmatrix} 2 \\ 4 \\ -1 \end{pmatrix}, \quad T \begin{pmatrix} 0 \\ 1 \\ 0 \end{pmatrix} = \begin{pmatrix} 1 \\ 3 \\ -2 \end{pmatrix}, \quad T \begin{pmatrix} 0 \\ 0 \\ 1 \end{pmatrix} = \begin{pmatrix} 0 \\ -2 \\ 2 \end{pmatrix}
$$

\text{Find:}

\subsection*{(a)}
$$
T \begin{pmatrix} 0 \\ 3 \\ -1 \end{pmatrix}
$$

\subsection*{(b)}
$$
T \begin{pmatrix} 2 \\ -1 \\ 0 \end{pmatrix}
$$
