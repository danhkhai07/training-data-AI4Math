\section*{Problem}

\text{Find the standard matrix of the following linear transformations $T$:}

\subsection*{(a)}
$$
T: \mathbf{R}^2 \to \mathbf{R}^2, \quad T \begin{pmatrix} x \\ y \end{pmatrix} = \begin{pmatrix} x + 2y \\ x - 2y \end{pmatrix}
$$

\subsection*{(b)}
$$
T: \mathbf{R}^3 \to \mathbf{R}^3, \quad T \begin{pmatrix} x \\ y \\ z \end{pmatrix} = \begin{pmatrix} 2x - 3y \\ x - y \\ z \end{pmatrix}
$$

\subsection*{(c)}
$$
T: \mathbf{R}^3 \to \mathbf{R}^3, \quad T \begin{pmatrix} x \\ y \\ z \end{pmatrix} = \begin{pmatrix} 0 \\ 0 \\ 0 \end{pmatrix}
$$
