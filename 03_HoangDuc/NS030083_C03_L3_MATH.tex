\section*{Problem}

\text{Let $\mathbf{A}$ be an $n \times n$ square matrix where $n$ is an odd positive integer. Suppose $\mathbf{A}$ is skew-symmetric (i.e., $\mathbf{A}^{\mathbf{T}} = -\mathbf{A}$). Prove that $\det(\mathbf{A}) = 0$.}
