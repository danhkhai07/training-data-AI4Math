\section*{Problem}

\text{Find the standard matrix of the following linear transformations $T$:}

\subsection*{(a)}
\text{$T: \mathbf{R}^2 \to \mathbf{R}^2$ is a rotation (about the origin) by an angle of $3\pi/2$ radians (counter-clockwise).}

\subsection*{(b)}
\text{$T: \mathbf{R}^2 \to \mathbf{R}^2$ first reflects points through the $x_2$-axis, and then reflects points through the line $x_2 = x_1$. Show that this transformation is simply a rotation about the origin. What is the angle of rotation?}

\subsection*{(c)}
\text{$T: \mathbf{R}^2 \to \mathbf{R}^2$ first transforms $\mathbf{e}_2$ into $\mathbf{e}_2 - 3\mathbf{e}_1$ (keeping $\mathbf{e}_1$ fixed), and then reflects points through the line $x_2 = -x_1$.}

\subsection*{(d)}
\text{$T: \mathbf{R}^2 \to \mathbf{R}^2$ first reflects points through the $x_2$-axis, and then rotates the points by an angle of $3\pi/2$ radians.}

\subsection*{(e)}
\text{$T: \mathbf{R}^2 \to \mathbf{R}^2$ first reflects points through the $x_1$-axis, and then reflects points through the $x_2$-axis. Show that $T$ can also be described as a linear transformation that rotates about the origin. What is the angle of rotation?}
