\section*{Problem}

\text{Write the system of linear equations that is equivalent to the given vector equation:}

\subsection*{(a)}
$$
x_1 \begin{pmatrix} 4 \\ -3 \\ 2 \end{pmatrix} + x_2 \begin{pmatrix} -8 \\ 7 \\ 0 \end{pmatrix} = \begin{pmatrix} 9 \\ -6 \\ -5 \end{pmatrix}
$$

\subsection*{(b)}
$$
x_1 \begin{pmatrix} -2 \\ 3 \end{pmatrix} + x_2 \begin{pmatrix} 8 \\ 5 \end{pmatrix} + x_3 \begin{pmatrix} 1 \\ -6 \end{pmatrix} = \begin{pmatrix} 0 \\ 0 \end{pmatrix}
$$
