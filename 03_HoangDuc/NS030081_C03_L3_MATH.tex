\section*{Problem}

\text{Find $x$ such that:}

\subsection*{(a)}
$$
\begin{vmatrix}
x + 3 & 1 \\
2 & x + 2
\end{vmatrix}
= 0
$$

\subsection*{(b)}
$$
\begin{vmatrix}
x - 2 & -1 \\
-3 & x
\end{vmatrix}
= 0
$$

\subsection*{(c)}
$$
\begin{vmatrix}
x & 2 & 0 \\
0 & x + 1 & 2 \\
0 & 1 & x
\end{vmatrix}
= 0
$$

\subsection*{(d)}
$$
\begin{vmatrix}
x & 0 & 1 \\
0 & x & 3 \\
2 & 2 & x - 2
\end{vmatrix}
= 0
$$
