\section*{Problem}

\text{Perform appropriate operations to answer the following questions for each matrix $\mathbf{A}$:}

\subsection*{(a) Matrix $\mathbf{A}_1$ and Questions}
$$
\mathbf{A}_1 = \begin{pmatrix}
1 & 3 & 0 & 3 \\
-1 & -1 & -1 & 1 \\
0 & -4 & 2 & -8 \\
2 & 0 & 3 & -1
\end{pmatrix}
$$

\begin{itemize}
    \item \text{How many rows of $\mathbf{A}_1$ contain a pivot position?}
    \item \text{Does the equation $\mathbf{A}_1\mathbf{x} = \mathbf{b}$ have a solution for every $\mathbf{b} \in \mathbf{R}^4$?}
\end{itemize}

\subsection*{(b) Matrix $\mathbf{A}_2$ and Questions}
$$
\mathbf{A}_2 = \begin{pmatrix}
1 & 3 & -2 & 2 \\
0 & 1 & 1 & -5 \\
1 & 2 & -3 & 7 \\
-2 & -8 & 2 & -1
\end{pmatrix}
$$

\begin{itemize}
    \item \text{Can every vector in $\mathbf{R}^4$ be written as a linear combination of the columns of $\mathbf{A}_2$?}
    \item \text{Do the columns of $\mathbf{A}_2$ span $\mathbf{R}^4$?}
\end{itemize}
