\section*{Problem}

\text{Prove that $T$ is a linear transformation by finding its standard matrix $\mathbf{A}$:}

\subsection*{(a)}
$$
T: \mathbf{R}^4 \to \mathbf{R}^4, \quad T \begin{pmatrix} x_1 \\ x_2 \\ x_3 \\ x_4 \end{pmatrix} = \begin{pmatrix} 0 \\ x_1 + x_2 \\ x_2 + x_3 \\ x_3 + x_4 \end{pmatrix}
$$

\subsection*{(b)}
$$
T: \mathbf{R}^2 \to \mathbf{R}^4, \quad T \begin{pmatrix} x_1 \\ x_2 \end{pmatrix} = \begin{pmatrix} 2x_2 - 3x_1 \\ x_1 - 4x_2 \\ 0 \\ x_2 \end{pmatrix}
$$

\subsection*{(c)}
$$
T: \mathbf{R}^3 \to \mathbf{R}^2, \quad T \begin{pmatrix} x_1 \\ x_2 \\ x_3 \end{pmatrix} = \begin{pmatrix} x_1 - 5x_2 + 4x_3 \\ x_2 - 6x_3 \end{pmatrix}
$$

\subsection*{(d)}
$$
T: \mathbf{R}^4 \to \mathbf{R}^1, \quad T \begin{pmatrix} x_1 \\ x_2 \\ x_3 \\ x_4 \end{pmatrix} = 2x_1 + 3x_3 - 4x_4
$$
