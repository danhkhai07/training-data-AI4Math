\section*{Problem}

\text{Find the standard matrix of the linear transformation $T = T_2 \circ T_1$ for the following cases:}

\subsection*{(a)}
$$
T_1: \mathbf{R}^2 \to \mathbf{R}^2, \quad T_1 \begin{pmatrix} x \\ y \end{pmatrix} = \begin{pmatrix} x - 2y \\ 2x + 3y \end{pmatrix}
$$
$$
T_2: \mathbf{R}^2 \to \mathbf{R}^2, \quad T_2 \begin{pmatrix} x \\ y \end{pmatrix} = \begin{pmatrix} 2x \\ x - y \end{pmatrix}
$$

\subsection*{(b)}
$$
T_1: \mathbf{R}^2 \to \mathbf{R}^3, \quad T_1 \begin{pmatrix} x \\ y \end{pmatrix} = \begin{pmatrix} x - 2y \\ x + y \\ x - y \end{pmatrix}
$$
$$
T_2: \mathbf{R}^3 \to \mathbf{R}^2, \quad T_2 \begin{pmatrix} x \\ y \\ z \end{pmatrix} = \begin{pmatrix} x - 3y \\ 3x + z \end{pmatrix}
$$

\subsection*{(c)}
$$
T_1: \mathbf{R}^3 \to \mathbf{R}^2, \quad T_1 \begin{pmatrix} x \\ y \\ z \end{pmatrix} = \begin{pmatrix} x - 3y \\ 3x + z \end{pmatrix}
$$
$$
T_2: \mathbf{R}^2 \to \mathbf{R}^3, \quad T_2 \begin{pmatrix} x \\ y \end{pmatrix} = \begin{pmatrix} x - 2y \\ x + y \\ x - y \end{pmatrix}
$$
