\section*{Problem}

\text{Solve the following systems of linear equations using Gaussian elimination:}

\subsection*{(a)}
\text{System of 3 equations with 2 variables:}
$$
\begin{cases}
-3x + 5y = -22 \\
3x + 4y = 4 \\
4x - 8y = 32
\end{cases}
$$

\subsection*{(b)}
\text{System of 3 equations with 3 variables:}
$$
\begin{cases}
x - 3z = -2 \\
3x + y - 2z = 5 \\
2x + 2y + z = 4
\end{cases}
$$

\subsection*{(c)}
\text{System of 3 equations with 3 variables:}
$$
\begin{cases}
2x - y + 3z = 24 \\
2y - z = 14 \\
7x - 5y = 6
\end{cases}
$$

\subsection*{(d)}
\text{System of 4 equations with 4 variables:}
$$
\begin{cases}
2x + y - z + 2w = -6 \\
3x + 4y + w = 1 \\
x + 5y + 2z + 6w = -3 \\
5x + 2y - z - w = 3
\end{cases}
$$
