\section*{Problem}

\text{Let $\mathbf{A} = \begin{pmatrix} 1 & 1/3 \\ a & b \end{pmatrix}$ and $\mathbf{I} = \begin{pmatrix} 1 & 0 \\ 0 & 1 \end{pmatrix}$. Find the values of $a$ and $b$ such that:}

\subsection*{(a)}
$$
\mathbf{A}^2 = \mathbf{I}
$$

\subsection*{(b)}
$$
\mathbf{A} = -\mathbf{I}
$$
