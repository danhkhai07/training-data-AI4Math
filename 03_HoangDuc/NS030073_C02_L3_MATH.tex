\section*{Problem}

\text{Prove that $T$ is invertible and find a formula for $T^{-1}$ for the linear transformation $T: \mathbf{R}^2 \to \mathbf{R}^2$ given by:}

\subsection*{(a)}
$$
T \begin{pmatrix} x_1 \\ x_2 \end{pmatrix} = \begin{pmatrix} -9x_1 + 7x_2 \\ 4x_1 - 3x_2 \end{pmatrix}
$$

\subsection*{(b)}
$$
T \begin{pmatrix} x_1 \\ x_2 \end{pmatrix} = \begin{pmatrix} 6x_1 - 8x_2 \\ -5x_1 + 7x_2 \end{pmatrix}
$$
