\section*{Problem}

\text{Use Cramer's Rule to solve the following systems of linear equations:}

\subsection*{(a)}
$$
\begin{cases}
x + 2y = 5 \\
-x + y = 1
\end{cases}
$$

\subsection*{(b)}
$$
\begin{cases}
4x - y - z = 1 \\
2x + 2y + 3z = 10 \\
5x - 2y - 2z = -1
\end{cases}
$$

\subsection*{(c)}
$$
\begin{cases}
4x - 2y + 3z = -2 \\
2x + 2y + 5z = 16 \\
8x - 5y - 2z = 4
\end{cases}
$$

\subsection*{(d)}
$$
\begin{cases}
3x + 3y + 5z = 1 \\
3x + 5y + 9z = 2 \\
5x + 9y + 17z = 4
\end{cases}
$$

\subsection*{(e)}
$$
\begin{cases}
2x + 3y + 5z = 4 \\
3x + 5y + 9z = 7 \\
5x + 9y + 17z = 13
\end{cases}
$$

\subsection*{(f)}
$$
\begin{cases}
2x + y + 2z = 6 \\
-x + 2y - 3z = 0 \\
3x + 2y - z = 6
\end{cases}
$$
