\section*{Problem}

\text{Calculate the determinant of the following matrices by using elementary row or column operations:}

\subsection*{(a)}
$$
\begin{pmatrix}
1 & 7 & -3 \\
1 & 3 & 1 \\
4 & 8 & 1
\end{pmatrix}
$$

\subsection*{(b)}
$$
\begin{pmatrix}
1 & 1 & 1 \\
2 & -1 & -2 \\
1 & -2 & -1
\end{pmatrix}
$$

\subsection*{(c)}
$$
\begin{pmatrix}
3 & -1 & -3 \\
-1 & -4 & -2 \\
3 & -1 & -1
\end{pmatrix}
$$

\subsection*{(d)}
$$
\begin{pmatrix}
4 & 5 & -2 \\
3 & 4 & 3 \\
-2 & 1 & 4
\end{pmatrix}
$$

\subsection*{(e)}
$$
\begin{pmatrix}
4 & -7 & 9 & 1 \\
6 & 2 & 7 & 0 \\
3 & 6 & -3 & 3 \\
0 & 7 & 4 & -1
\end{pmatrix}
$$

\subsection*{(f)}
$$
\begin{pmatrix}
9 & -4 & 2 & 5 \\
2 & 7 & 6 & -5 \\
4 & 1 & -2 & 0 \\
7 & 3 & 4 & 10
\end{pmatrix}
$$
