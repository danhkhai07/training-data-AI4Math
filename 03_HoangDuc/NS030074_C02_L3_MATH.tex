\section*{Problem}

\text{Which of the following linear transformations is an isomorphism? If it is, find the inverse transformation $T^{-1}$:}

\subsection*{(a)}
$$
T: \mathbf{R}^3 \to \mathbf{R}^3, \quad T \begin{pmatrix} x \\ y \\ z \end{pmatrix} = \begin{pmatrix} x \\ x + y \\ x + y + z \end{pmatrix}
$$

\subsection*{(b)}
$$
T: \mathbf{R}^3 \to \mathbf{R}^3, \quad T \begin{pmatrix} x \\ y \\ z \end{pmatrix} = \begin{pmatrix} x + y \\ y + z \\ x + z \end{pmatrix}
$$

\subsection*{(c)}
$$
T: \mathbf{R}^4 \to \mathbf{R}^4, \quad T \begin{pmatrix} x_1 \\ x_2 \\ x_3 \\ x_4 \end{pmatrix} = \begin{pmatrix} x_1 - 2x_2 \\ x_2 \\ x_3 + x_4 \\ x_3 \end{pmatrix}
$$

\subsection*{(d)}
$$
T: \mathbf{R}^4 \to \mathbf{R}^4, \quad T \begin{pmatrix} x_1 \\ x_2 \\ x_3 \\ x_4 \end{pmatrix} = \begin{pmatrix} x_4 \\ x_3 \\ x_2 \\ x_1 \end{pmatrix}
$$
