\section*{Problem}

\text{Check whether the following points are coplanar:}

\subsection*{(a)}
$$
\begin{pmatrix} -4 \\ 1 \\ 0 \end{pmatrix}, \quad \begin{pmatrix} 0 \\ 1 \\ 2 \end{pmatrix}, \quad \begin{pmatrix} 4 \\ 3 \\ -1 \end{pmatrix}, \quad \text{and} \quad \begin{pmatrix} 0 \\ 0 \\ 1 \end{pmatrix}
$$

\subsection*{(b)}
$$
\begin{pmatrix} 1 \\ 2 \\ 3 \end{pmatrix}, \quad \begin{pmatrix} -1 \\ 0 \\ 1 \end{pmatrix}, \quad \begin{pmatrix} 0 \\ -2 \\ -5 \end{pmatrix}, \quad \text{and} \quad \begin{pmatrix} 2 \\ 6 \\ 11 \end{pmatrix}
$$

\subsection*{(c)}
$$
\begin{pmatrix} 0 \\ 0 \\ -1 \end{pmatrix}, \quad \begin{pmatrix} 0 \\ -1 \\ 0 \end{pmatrix}, \quad \begin{pmatrix} 1 \\ 1 \\ 0 \end{pmatrix}, \quad \text{and} \quad \begin{pmatrix} 2 \\ 1 \\ 2 \end{pmatrix}
$$

\subsection*{(d)}
$$
\begin{pmatrix} 1 \\ 2 \\ 7 \end{pmatrix}, \quad \begin{pmatrix} -3 \\ 6 \\ 6 \end{pmatrix}, \quad \begin{pmatrix} 4 \\ 4 \\ 2 \end{pmatrix}, \quad \text{and} \quad \begin{pmatrix} 3 \\ 3 \\ 4 \end{pmatrix}
$$
