\section*{Problem Statement}

\text{Let $\mathbf{v}_1$, $\mathbf{v}_2$, and $\mathbf{v}_3$ be the following vectors in $\mathbf{R}^4$:}
$$
\mathbf{v}_1 = \begin{pmatrix} 1 \\ 0 \\ -1 \\ 0 \end{pmatrix}, \quad \mathbf{v}_2 = \begin{pmatrix} 0 \\ -1 \\ 0 \\ 1 \end{pmatrix}, \quad \mathbf{v}_3 = \begin{pmatrix} 1 \\ 0 \\ 0 \\ -1 \end{pmatrix}
$$

\text{Does the set $\{\mathbf{v}_1, \mathbf{v}_2, \mathbf{v}_3\}$ span $\mathbf{R}^4$? Explain why or why not.}

\text{Note: For a set of vectors to span $\mathbf{R}^n$, it must contain at least $n$ vectors.}
