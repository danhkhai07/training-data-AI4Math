\section*{Problem}

\text{Calculate the volume of the parallelepiped with one vertex at the origin and three adjacent vertices at:}

\subsection*{(a)}
$$
\begin{pmatrix} 1 \\ 0 \\ -6 \end{pmatrix}, \quad \begin{pmatrix} 1 \\ 3 \\ 5 \end{pmatrix}, \quad \text{and} \quad \begin{pmatrix} 6 \\ 7 \\ 0 \end{pmatrix}
$$

\subsection*{(b)}
$$
\begin{pmatrix} 1 \\ 5 \\ 0 \end{pmatrix}, \quad \begin{pmatrix} -3 \\ 0 \\ 3 \end{pmatrix}, \quad \text{and} \quad \begin{pmatrix} -1 \\ 4 \\ -1 \end{pmatrix}
$$
