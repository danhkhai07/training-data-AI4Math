\section*{Problem}

\text{Write the vector equation that is equivalent to the given system of linear equations:}

\subsection*{(a)}
$$
\begin{cases}
x_2 + 5x_3 = 0 \\
4x_1 + 6x_2 - x_3 = 0 \\
-x_1 + 3x_2 - 8x_3 = 0
\end{cases}
$$

\subsection*{(b)}
$$
\begin{cases}
4x_1 + x_2 + 3x_3 = 9 \\
x_1 - 7x_2 - 2x_3 = 2 \\
8x_1 + 6x_2 - 5x_3 = 15
\end{cases}
$$
