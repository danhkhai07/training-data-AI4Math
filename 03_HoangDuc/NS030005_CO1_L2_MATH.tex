\section*{Problem}

\text{Discuss the number of solutions for the following systems of equations based on the parameter $k$:}

\subsection*{(a)}
\text{System of 2 equations with 2 variables:}
$$
\begin{cases}
4x + ky = 6 \\
kx + y = -3
\end{cases}
$$

\subsection*{(b)}
\text{System of 2 equations with 3 variables:}
$$
\begin{cases}
x + 2y + kz = 6 \\
3x + 6y + 8z = 4
\end{cases}
$$

\subsection*{(c)}
\text{System of 3 equations with 3 variables:}
$$
\begin{cases}
x + y + kz = 3 \\
x + ky + z = 2 \\
kx + y + z = 1
\end{cases}
$$

\subsection*{(d)}
\text{System of 3 equations with 3 variables:}
$$
\begin{cases}
x - y + 3z = -2 \\
2x + 7z = -3 \\
-3x + y + (k - 8)z = 5
\end{cases}
$$

\subsection*{(e)}
\text{System of 3 equations with 3 variables:}
$$
\begin{cases}
(k - 1)x + 3y + 3z = 3 \\
6x + 6y + 12z = 13 \\
12x + 9y - z = 2
\end{cases}
$$
