\section*{Problem}

\text{Let $\mathbf{A} = \begin{pmatrix} -2 & -7 & -9 \\ 2 & 5 & 6 \\ 1 & 3 & 4 \end{pmatrix}$. Find the third column of $\mathbf{A}^{-1}$ without computing the other columns.}

\section*{Solution}

\text{Let $\mathbf{B} = \mathbf{A}^{-1}$, and let $\mathbf{b}_3$ be the third column of $\mathbf{B}$. Since $\mathbf{A}\mathbf{B} = \mathbf{I}$, the third column of the product $\mathbf{A}\mathbf{B}$ must equal the third column of $\mathbf{I}$, which is the standard vector $\mathbf{e}_3 = \begin{pmatrix} 0 \\ 0 \\ 1 \end{pmatrix}$.}
\text{We must solve the system $\mathbf{A}\mathbf{b}_3 = \mathbf{e}_3$:}
$$
\mathbf{A}\mathbf{b}_3 = \begin{pmatrix} -2 & -7 & -9 \\ 2 & 5 & 6 \\ 1 & 3 & 4 \end{pmatrix} \begin{pmatrix} x_1 \\ x_2 \\ x_3 \end{pmatrix} = \begin{pmatrix} 0 \\ 0 \\ 1 \end{pmatrix}
$$

\text{We solve the augmented matrix $\left[ \mathbf{A} \mid \mathbf{e}_3 \right]$ using row reduction:}
$$
\begin{pmatrix}
-2 & -7 & -9 & \mid & 0 \\
2 & 5 & 6 & \mid & 0 \\
1 & 3 & 4 & \mid & 1
\end{pmatrix}
\xrightarrow{\text{R}_1 \leftrightarrow \text{R}_3}
\begin{pmatrix}
1 & 3 & 4 & \mid & 1 \\
2 & 5 & 6 & \mid & 0 \\
-2 & -7 & -9 & \mid & 0
\end{pmatrix}
$$
$$
\xrightarrow{\substack{\text{R}_2 \to \text{R}_2 - 2\text{R}_1 \\ \text{R}_3 \to \text{R}_3 + 2\text{R}_1}}
\begin{pmatrix}
1 & 3 & 4 & \mid & 1 \\
0 & -1 & -2 & \mid & -2 \\
0 & -1 & -1 & \mid & 2
\end{pmatrix}
\xrightarrow{\text{R}_2 \to -\text{R}_2}
\begin{pmatrix}
1 & 3 & 4 & \mid & 1 \\
0 & 1 & 2 & \mid & 2 \\
0 & -1 & -1 & \mid & 2
\end{pmatrix}
$$
$$
\xrightarrow{\text{R}_3 \to \text{R}_3 + \text{R}_2}
\begin{pmatrix}
1 & 3 & 4 & \mid & 1 \\
0 & 1 & 2 & \mid & 2 \\
0 & 0 & 1 & \mid & 4
\end{pmatrix}
\xrightarrow{\substack{\text{R}_2 \to \text{R}_2 - 2\text{R}_3 \\ \text{R}_1 \to \text{R}_1 - 4\text{R}_3}}
\begin{pmatrix}
1 & 3 & 0 & \mid & -15 \\
0 & 1 & 0 & \mid & -6 \\
0 & 0 & 1 & \mid & 4
\end{pmatrix}
$$
$$
\xrightarrow{\text{R}_1 \to \text{R}_1 - 3\text{R}_2}
\begin{pmatrix}
1 & 0 & 0 & \mid & 3 \\
0 & 1 & 0 & \mid & -6 \\
0 & 0 & 1 & \mid & 4
\end{pmatrix}
$$

\text{The third column of $\mathbf{A}^{-1}$ is the solution vector:}
$$
\mathbf{A}^{-1}\mathbf{e}_3 = \mathbf{b}_3 = \begin{pmatrix} 3 \\ -6 \\ 4 \end{pmatrix}
$$
