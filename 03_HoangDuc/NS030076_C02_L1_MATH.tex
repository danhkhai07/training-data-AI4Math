\section*{Problem}

\text{Find formulas for $\mathbf{X}$, $\mathbf{Y}$, and $\mathbf{Z}$ in terms of $\mathbf{A}$, $\mathbf{B}$, and $\mathbf{C}$, assuming all necessary inverses exist:}

\subsection*{(a)}
$$
\begin{pmatrix} \mathbf{X} & \mathbf{0} \\ \mathbf{Y} & \mathbf{Z} \end{pmatrix} \begin{pmatrix} \mathbf{A} & \mathbf{0} \\ \mathbf{B} & \mathbf{C} \end{pmatrix} = \begin{pmatrix} \mathbf{I} & \mathbf{0} \\ \mathbf{0} & \mathbf{I} \end{pmatrix}
$$

\subsection*{(b)}
$$
\begin{pmatrix} \mathbf{X} & \mathbf{0} & \mathbf{0} \\ \mathbf{Y} & \mathbf{0} & \mathbf{I} \end{pmatrix} \begin{pmatrix} \mathbf{A} & \mathbf{Z} \\ \mathbf{0} & \mathbf{0} \\ \mathbf{B} & \mathbf{I} \end{pmatrix} = \begin{pmatrix} \mathbf{I} & \mathbf{0} \\ \mathbf{0} & \mathbf{I} \end{pmatrix}
$$
