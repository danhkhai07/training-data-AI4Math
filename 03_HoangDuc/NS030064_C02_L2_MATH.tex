\section*{Problem}

\text{Prove that matrix $\mathbf{B}$ is the inverse of matrix $\mathbf{A}$ for the following pairs:}

\subsection*{(a)}
$$
\mathbf{A} = \begin{pmatrix} 1 & 2 \\ 3 & 4 \end{pmatrix}, \quad \mathbf{B} = \begin{pmatrix} -2 & 1 \\ 3/2 & -1/2 \end{pmatrix}
$$

\subsection*{(b)}
$$
\mathbf{A} = \begin{pmatrix} 1 & -1 \\ 2 & 3 \end{pmatrix}, \quad \mathbf{B} = \frac{1}{5} \begin{pmatrix} 3 & 1 \\ -2 & 1 \end{pmatrix}
$$

\subsection*{(c)}
$$
\mathbf{A} = \begin{pmatrix} -2 & 2 & 3 \\ 1 & -1 & 0 \\ 0 & 1 & 4 \end{pmatrix}, \quad \mathbf{B} = \frac{1}{3} \begin{pmatrix} -4 & -5 & 3 \\ -4 & -8 & 3 \\ 1 & 2 & 0 \end{pmatrix}
$$

\subsection*{(d)}
$$
\mathbf{A} = \begin{pmatrix} 2 & -17 & 11 \\ -1 & 11 & -7 \\ 0 & 3 & -2 \end{pmatrix}, \quad \mathbf{B} = \begin{pmatrix} 1 & 1 & 2 \\ 2 & 4 & -3 \\ 3 & 6 & -5 \end{pmatrix}
$$
