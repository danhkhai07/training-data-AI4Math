\section*{Problem}

\text{Find $(\mathbf{A}\mathbf{B})^{-1}$, $(\mathbf{A}^T)^{-1}$, $\mathbf{A}^{-2}$, and $(2\mathbf{B})^{-1}$ for the following cases, given the inverse matrices $\mathbf{A}^{-1}$ and $\mathbf{B}^{-1}$:}

\subsection*{(a)}
$$
\mathbf{A}^{-1} = \begin{pmatrix} 2 & 5 \\ -7 & 6 \end{pmatrix}, \quad \mathbf{B}^{-1} = \begin{pmatrix} 7 & -3 \\ 2 & 0 \end{pmatrix}
$$

\subsection*{(b)}
$$
\mathbf{A}^{-1} = \begin{pmatrix} -2/7 & 1/7 \\ 3/7 & 2/7 \end{pmatrix}, \quad \mathbf{B}^{-1} = \begin{pmatrix} 5/11 & 2/11 \\ 3/11 & -1/11 \end{pmatrix}
$$

\subsection*{(c)}
$$
\mathbf{A}^{-1} = \begin{pmatrix} 1 & -1/2 & 3/4 \\ 3/2 & 1/2 & -2 \\ 1/4 & 1 & 1/2 \end{pmatrix}, \quad \mathbf{B}^{-1} = \begin{pmatrix} 2 & 4 & 5 \\ 2 & -3/4 & 2 \\ 1/4 & 1/4 & 1/2 \end{pmatrix}
$$

\subsection*{(d)}
$$
\mathbf{A}^{-1} = \begin{pmatrix} 1 & -4 & 2 \\ 0 & 1 & 3 \\ 4 & 2 & 1 \end{pmatrix}, \quad \mathbf{B}^{-1} = \begin{pmatrix} 6 & 5 & -3 \\ -2 & 4 & -1 \\ 1 & 3 & 4 \end{pmatrix}
$$
