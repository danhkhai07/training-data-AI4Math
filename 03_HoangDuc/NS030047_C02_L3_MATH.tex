\section*{Problem}

\text{Let $T: \mathbf{R}^2 \to \mathbf{R}^3$ be a linear transformation defined by the matrix $\mathbf{A}$.}
$$
\mathbf{A} = \begin{pmatrix}
1 & 2 \\
-2 & 4 \\
-2 & 2
\end{pmatrix}
$$

\text{Find $T \begin{pmatrix} 2 \\ 4 \end{pmatrix}$ and $T^{-1} \begin{pmatrix} -1 \\ 2 \\ 2 \end{pmatrix}$.}
