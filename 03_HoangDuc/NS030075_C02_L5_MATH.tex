\section*{Problem}

\text{Let $\mathbf{A}$ and $\mathbf{B}$ be two invertible real matrices. Suppose $\mathbf{A}^5 = \mathbf{I}$, $\mathbf{A}\mathbf{B}^2 = \mathbf{B}\mathbf{A}$, and $\mathbf{B} \neq \mathbf{I}$. Find the smallest positive integer $k$ such that $\mathbf{B}^k = \mathbf{I}$.}
