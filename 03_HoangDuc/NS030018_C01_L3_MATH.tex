\section*{Problem}

\text{Let $\mathbf{A}$ and $\mathbf{b}$ be defined below. Let the columns of $\mathbf{A}$ be $\mathbf{a}_1, \mathbf{a}_2, \mathbf{a}_3$,} \\
\text{and set $W = \text{Span}\{\mathbf{a}_1, \mathbf{a}_2, \mathbf{a}_3\}$.}

$$
\mathbf{A} = \begin{pmatrix} 1 & 0 & -4 \\ 0 & 3 & -2 \\ -2 & 6 & 3 \end{pmatrix} \quad \text{and} \quad \mathbf{b} = \begin{pmatrix} 4 \\ 1 \\ -4 \end{pmatrix}
$$

\subsection*{(a)}
\text{Is $\mathbf{b}$ in $\{\mathbf{a}_1, \mathbf{a}_2, \mathbf{a}_3\}$? How many vectors are in $\{\mathbf{a}_1, \mathbf{a}_2, \mathbf{a}_3\}$?}

\subsection*{(b)}
\text{Is $\mathbf{b}$ in $W$? How many vectors are in $W$?}

\subsection*{(c)}
\text{Show that $\mathbf{a}_1 \in W$.}
