\section*{Problem}

\text{Determine if the columns of the following matrix $\mathbf{A}$ span $\mathbf{R}^4$:}
$$
\mathbf{A} = \begin{pmatrix}
8 & 11 & -6 & -7 & 13 \\
-7 & -8 & 5 & 6 & -9 \\
11 & 7 & -7 & -9 & -6 \\
-3 & 4 & 1 & 8 & 7
\end{pmatrix}
$$

\text{Find a column that can be removed so that the remaining columns still span $\mathbf{R}^4$.} \\
\text{Can more than one column be removed?}
