\section*{Problem}

\text{Let $\mathbf{u} = \begin{pmatrix} 7 \\ 2 \\ 5 \end{pmatrix}$, $\mathbf{v} = \begin{pmatrix} 3 \\ 1 \\ 3 \end{pmatrix}$, and $\mathbf{w} = \begin{pmatrix} 6 \\ 1 \\ 0 \end{pmatrix}$.} \\
\text{First, show that $3\mathbf{u} - 5\mathbf{v} - \mathbf{w} = \mathbf{0}$.} \\
\text{Then, use this fact (and no Gaussian elimination) to find $x_1$ and $x_2$ that satisfy the equation:}
$$
\begin{pmatrix}
7 & 3 \\
2 & 1 \\
5 & 3
\end{pmatrix}
\begin{pmatrix}
x_1 \\ x_2
\end{pmatrix}
=
\begin{pmatrix}
6 \\ 1 \\ 0
\end{pmatrix}
$$
