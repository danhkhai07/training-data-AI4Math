\section*{Problem}

\text{For each of the following sets of vectors, determine whether the set is linearly}\\
\text{independent or linearly dependent:}

\subsection*{(a)}
$$
\left\{ \begin{pmatrix} 0 \\ 0 \end{pmatrix}, \begin{pmatrix} 1 \\ 2 \end{pmatrix} \right\}
$$

\subsection*{(b)}
$$
\left\{ \begin{pmatrix} 1 \\ 0 \end{pmatrix}, \begin{pmatrix} 1 \\ 1 \end{pmatrix}, \begin{pmatrix} 2 \\ -1 \end{pmatrix} \right\}
$$

\subsection*{(c)}
$$
\left\{ \begin{pmatrix} 1 \\ -4 \\ 1 \end{pmatrix}, \begin{pmatrix} 6 \\ 3 \\ 2 \end{pmatrix} \right\}
$$

\subsection*{(d)}
$$
\left\{ \begin{pmatrix} 6 \\ 2 \\ 1 \end{pmatrix}, \begin{pmatrix} -1 \\ 3 \\ 2 \end{pmatrix} \right\}
$$

\subsection*{(e)}
$$
\left\{ \begin{pmatrix} 1 \\ 1 \\ 1 \end{pmatrix}, \begin{pmatrix} 2 \\ 2 \\ 2 \end{pmatrix}, \begin{pmatrix} 1 \\ 2 \\ 3 \end{pmatrix} \right\}
$$

\subsection*{(f)}
$$
\left\{ \begin{pmatrix} -4 \\ -3 \\ 4 \end{pmatrix}, \begin{pmatrix} 1 \\ -2 \\ 3 \end{pmatrix}, \begin{pmatrix} 6 \\ 0 \\ 0 \end{pmatrix} \right\}
$$

\subsection*{(g)}
$$
\left\{ \begin{pmatrix} 1 \\ 0 \\ 0 \end{pmatrix}, \begin{pmatrix} 0 \\ 4 \\ 0 \end{pmatrix}, \begin{pmatrix} 0 \\ 0 \\ -6 \end{pmatrix}, \begin{pmatrix} 1 \\ 5 \\ -3 \end{pmatrix} \right\}
$$

\subsection*{(h)}
$$
\left\{
\begin{pmatrix} 0 \\ 0 \\ 0 \\ 1 \end{pmatrix},
\begin{pmatrix} 0 \\ 0 \\ 1 \\ 1 \end{pmatrix},
\begin{pmatrix} 0 \\ 1 \\ 1 \\ 1 \end{pmatrix},
\begin{pmatrix} 1 \\ 1 \\ 1 \\ 1 \end{pmatrix}
\right\}
$$
