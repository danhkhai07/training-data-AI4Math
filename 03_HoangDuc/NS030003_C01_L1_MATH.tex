\section*{Problem}

\text{Solve the following system of equations in echelon form:}

\subsection*{(a)}
\text{Echelon system with 3 variables ($x, y, z$):}
$$
\begin{cases}
x - y = 4 \\
2y + z = 6 \\
3z = 6
\end{cases}
$$

\subsection*{(b)}
\text{Echelon system with 4 variables ($x_1, x_2, x_3, x_4$):}
$$
\begin{cases}
x_1 + x_2 + x_3 + x_4 = 10 \\
x_2 + 3x_3 + 4x_4 = 15
\end{cases}
$$
