\section*{Problem}

\text{Find a non-trivial linear combination equal to zero ($\mathbf{0}$) for each set of vectors:}

\subsection*{(a)}
$$
\left\{ \begin{pmatrix} 3 \\ 4 \end{pmatrix}, \begin{pmatrix} -1 \\ 1 \end{pmatrix}, \begin{pmatrix} 2 \\ 0 \end{pmatrix} \right\}
$$

\subsection*{(b)}
$$
\left\{ \begin{pmatrix} 2 \\ 4 \end{pmatrix}, \begin{pmatrix} -1 \\ -2 \end{pmatrix}, \begin{pmatrix} 0 \\ 6 \end{pmatrix} \right\}
$$

\subsection*{(c)}
$$
\left\{
\begin{pmatrix} 1 \\ 1 \\ 1 \end{pmatrix},
\begin{pmatrix} 1 \\ 1 \\ 0 \end{pmatrix},
\begin{pmatrix} 0 \\ 1 \\ 1 \end{pmatrix},
\begin{pmatrix} 0 \\ 0 \\ 1 \end{pmatrix}
\right\}
$$

\subsection*{(d)}
$$
\left\{
\begin{pmatrix} 1 \\ 2 \\ 3 \\ 4 \end{pmatrix},
\begin{pmatrix} 1 \\ 0 \\ 1 \\ 2 \end{pmatrix},
\begin{pmatrix} 1 \\ 4 \\ 5 \\ 6 \end{pmatrix}
\right\}
$$
