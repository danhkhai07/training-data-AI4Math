\section*{Problem}

\text{Write the following systems of linear equations in the matrix form $\mathbf{A}\mathbf{x} = \mathbf{b}$. Then, find the inverse matrix $\mathbf{A}^{-1}$ and use it to find the solution $\mathbf{x}$.}

\subsection*{(a)}
$$
\begin{cases}
5x + 4y = 2 \\
-x + y = -22
\end{cases}
$$

\subsection*{(b)}
$$
\begin{cases}
3x + 2y = 1 \\
x + 4y = -3
\end{cases}
$$

\subsection*{(c)}
$$
\begin{cases}
-x + y + 2z = 1 \\
2x + 3y + z = -2 \\
5x + 4y + 2z = 4
\end{cases}
$$

\subsection*{(d)}
$$
\begin{cases}
x + y + 2z = 0 \\
x - y + z = -1 \\
2x + y + z = 2
\end{cases}
$$
