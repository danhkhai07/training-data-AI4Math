\section*{Problem}

\text{Find the area of the image of the parallelogram $S$ determined by vectors $\mathbf{b}_1$ and $\mathbf{b}_2$ under the linear mapping $\mathbf{x} \mapsto \mathbf{A}\mathbf{x}$, where:}

\subsection*{(a)}
$$
\mathbf{b}_1 = \begin{pmatrix} -3 \\ 5 \end{pmatrix}, \quad \mathbf{b}_2 = \begin{pmatrix} -3 \\ 8 \end{pmatrix}, \quad \mathbf{A} = \begin{pmatrix} 3 & -4 \\ -4 & 6 \end{pmatrix}
$$

\subsection*{(b)}
$$
\mathbf{b}_1 = \begin{pmatrix} -3 \\ 5 \end{pmatrix}, \quad \mathbf{b}_2 = \begin{pmatrix} 0 \\ -3 \end{pmatrix}, \quad \mathbf{A} = \begin{pmatrix} 3 & 4 \\ -2 & -2 \end{pmatrix}
$$
