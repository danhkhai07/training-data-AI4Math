\section*{Problem}

\text{Which of the following mappings is a linear transformation:}

\subsection*{(a)}
$$
T: \mathbf{R}^2 \to \mathbf{R}^2, \quad T \begin{pmatrix} x \\ y \end{pmatrix} = \begin{pmatrix} x \\ 1 \end{pmatrix}
$$

\subsection*{(b)}
$$
T: \mathbf{R}^3 \to \mathbf{R}^3, \quad T \begin{pmatrix} x \\ y \\ z \end{pmatrix} = \begin{pmatrix} x + y \\ x - y \\ z \end{pmatrix}
$$

\subsection*{(c)}
$$
T: \mathbf{R}^3 \to \mathbf{R}^3, \quad T \begin{pmatrix} x \\ y \\ z \end{pmatrix} = \begin{pmatrix} x + 1 \\ y + 1 \\ z + 1 \end{pmatrix}
$$
