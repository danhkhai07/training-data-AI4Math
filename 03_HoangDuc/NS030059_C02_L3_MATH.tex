\section*{Problem}

\text{Find the value(s) of $m$ such that $\mathbf{A}^2 = \mathbf{I}$, where $\mathbf{I}$ is the $3 \times 3$ identity matrix and $\mathbf{A}$ is given by:}
$$
\mathbf{A} = \begin{pmatrix}
1 & m - 1 & 0 \\
0 & 0 & m \\
m^2 - m & m & 0
\end{pmatrix}
$$
