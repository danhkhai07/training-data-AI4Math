\section*{Problem}

\text{Determine whether the columns of the following matrices form a linearly independent set:}

\subsection*{(a)}
$$
\mathbf{A}_a = \begin{pmatrix}
0 & -8 & 5 \\
3 & -7 & 4 \\
-1 & 5 & -4 \\
1 & -3 & 2
\end{pmatrix}
$$

\subsection*{(b)}
$$
\mathbf{A}_b = \begin{pmatrix}
-4 & -3 & 0 \\
0 & -1 & 4 \\
1 & 0 & 3 \\
5 & 4 & 6
\end{pmatrix}
$$

\subsection*{(c)}
$$
\mathbf{A}_c = \begin{pmatrix}
1 & 4 & -3 & 0 \\
-2 & -7 & 5 & 1 \\
-4 & -5 & 7 & 5
\end{pmatrix}
$$

\subsection*{(d)}
$$
\mathbf{A}_d = \begin{pmatrix}
1 & -3 & 3 & -2 \\
-3 & 7 & -1 & 2 \\
0 & 1 & -4 & 3
\end{pmatrix}
$$
