\section*{Problem}

\text{Consider the following mappings:}

\subsection*{(a)}
$$
T_1: \mathbf{R}^3 \to \mathbf{R}^3, \quad T_1(\mathbf{v}) = -\mathbf{v}
$$

\subsection*{(b)}
$$
T_2: \mathbf{R}^3 \to \mathbf{R}^2, \quad T_2 \begin{pmatrix} x \\ y \\ z \end{pmatrix} = - \begin{pmatrix} x + 1 \\ y + z \end{pmatrix}
$$

\subsection*{(c)}
$$
T_3 \text{ is the orthogonal projection from } \mathbf{R}^3 \text{ onto the } x\text{-axis: } T_3 \begin{pmatrix} x \\ y \\ z \end{pmatrix} = \begin{pmatrix} x \\ 0 \\ 0 \end{pmatrix}
$$

\text{Which of these mappings are linear transformations? Explain your answers.} \\
\text{Find the standard matrix of the linear transformations.}
