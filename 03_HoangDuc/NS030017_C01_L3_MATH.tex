\section*{Problem Statement}

\text{Let $\mathbf{v}_1$ and $\mathbf{v}_2$ be two vectors and $\mathbf{y}$ be a third vector, defined as:}
$$
\mathbf{v}_1 = \begin{pmatrix} 1 \\ 0 \\ -4 \end{pmatrix}, \quad \mathbf{v}_2 = \begin{pmatrix} -5 \\ 1 \\ 7 \end{pmatrix}, \quad \mathbf{y} = \begin{pmatrix} h \\ -1 \\ -5 \end{pmatrix}
$$

\text{For what value(s) of the scalar $h$ is the vector $\mathbf{y}$ in the plane spanned by $\mathbf{v}_1$ and $\mathbf{v}_2$?}

\text{This condition is met if the vector equation $x_1\mathbf{v}_1 + x_2\mathbf{v}_2 = \mathbf{y}$ is consistent.}
