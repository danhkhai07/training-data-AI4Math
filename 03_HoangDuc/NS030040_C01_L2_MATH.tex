\section*{Problem}

\text{Find two $3 \times 2$ matrices, $\mathbf{A}$ and $\mathbf{B}$, such that:} \\
\begin{itemize}
    \item \text{The equation $\mathbf{A}\mathbf{x} = \mathbf{0}$ has only the trivial solution.}
    \item \text{The equation $\mathbf{B}\mathbf{x} = \mathbf{0}$ has a non-trivial solution.}
\end{itemize}
