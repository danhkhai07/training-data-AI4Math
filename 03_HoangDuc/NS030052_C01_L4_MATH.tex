\section*{Problem}

\text{Determine if $T$ is one-to-one (injective) and/or onto (surjective), where $T$ is the linear transformation whose standard matrix is given by:}

\subsection*{(a)}
$$
\mathbf{A}_a = \begin{pmatrix}
-5 & 10 & -5 & 4 \\
8 & 3 & -4 & 7 \\
4 & -9 & 5 & -3 \\
-3 & -2 & 5 & 4
\end{pmatrix}
$$

\subsection*{(b)}
$$
\mathbf{A}_b = \begin{pmatrix}
7 & 5 & 4 & -9 \\
10 & 6 & 16 & -4 \\
12 & 8 & 12 & 7 \\
-8 & -6 & -2 & 5
\end{pmatrix}
$$
