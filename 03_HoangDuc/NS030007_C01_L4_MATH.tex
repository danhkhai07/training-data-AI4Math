\section*{Problem}

\text{Find an equation relating $g$, $h$, and $k$ such that the augmented matrix below}\\
\text{corresponds to a consistent system of linear equations:}
$$
\left[
\begin{array}{ccc|c}
1 & -3 & 5 & g \\
0 & 2 & -3 & h \\
-3 & 5 & -9 & k
\end{array}
\right]
$$
