\section*{Problem}

\text{Determine if $\mathbf{b}$ is a linear combination of the vectors formed by the columns of the matrix $\mathbf{A}$, where:}

\subsection*{(a)}
$$
\mathbf{A} = \begin{pmatrix} -1 & 2 \\ 4 & 0 \end{pmatrix}, \quad \mathbf{b} = \begin{pmatrix} 3 \\ 4 \end{pmatrix}
$$

\subsection*{(b)}
$$
\mathbf{A} = \begin{pmatrix} -1 & 2 \\ 2 & -4 \end{pmatrix}, \quad \mathbf{b} = \begin{pmatrix} 2 \\ 4 \end{pmatrix}
$$

\subsection*{(c)}
$$
\mathbf{A} = \begin{pmatrix}
1 & 3 & 0 \\
-1 & 1 & 0 \\
0 & 1 & 1
\end{pmatrix}, \quad \mathbf{b} = \begin{pmatrix} 1 \\ 2 \\ -3 \end{pmatrix}
$$
