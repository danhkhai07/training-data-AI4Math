\section*{Problem}

\text{Solve the following systems of linear equations by reducing them to echelon form:}

\subsection*{(a)}
\text{System of 2 equations with 2 variables:}
$$
\begin{cases}
x + 2y = 7 \\
2x + y = 8
\end{cases}
$$

\subsection*{(b)}
\text{System of 3 equations with 2 variables:}
$$
\begin{cases}
x + 2y = 0 \\
x + y = 6 \\
3x - 2y = 8
\end{cases}
$$

\subsection*{(c)}
\text{System of 3 equations with 3 variables:}
$$
\begin{cases}
x + y + z = 2 \\
-x + 3y + 2z = 8 \\
4x + y = 4
\end{cases}
$$

\subsection*{(d)}
\text{System of 3 equations with 3 variables:}
$$
\begin{cases}
5x - 3y + 2z = 3 \\
2x + 4y - z = 7 \\
x - 11y + 4z = 3
\end{cases}
$$

\subsection*{(e)}
\text{System of 4 equations with 4 variables:}
$$
\begin{cases}
x + y + z + w = 6 \\
2x + 3y - w = 0 \\
-3x + 4y + z + 2w = 4 \\
x + 2y - z + w = 0
\end{cases}
$$

\subsection*{(f)}
\text{Homogeneous system of 3 equations with 3 variables:}
$$
\begin{cases}
4x + 3y + 17z = 0 \\
5x + 4y + 22z = 0 \\
4x + 2y + 19z = 0
\end{cases}
$$
