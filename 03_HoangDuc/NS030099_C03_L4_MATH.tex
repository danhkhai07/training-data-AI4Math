\section*{Problem}

\text{Calculate the volume of the tetrahedron with the following vertex coordinates:}

\subsection*{(a)}
$$
\begin{pmatrix} 1 \\ 1 \\ 1 \end{pmatrix}, \quad \begin{pmatrix} 0 \\ 0 \\ 0 \end{pmatrix}, \quad \begin{pmatrix} 2 \\ 1 \\ -1 \end{pmatrix}, \quad \text{and} \quad \begin{pmatrix} -1 \\ 1 \\ 2 \end{pmatrix}
$$

\subsection*{(b)}
$$
\begin{pmatrix} 3 \\ -1 \\ 1 \end{pmatrix}, \quad \begin{pmatrix} 4 \\ -4 \\ 4 \end{pmatrix}, \quad \begin{pmatrix} 1 \\ 1 \\ 1 \end{pmatrix}, \quad \text{and} \quad \begin{pmatrix} 0 \\ 0 \\ 1 \end{pmatrix}
$$
