\section*{Problem}

\text{For what value of $m$ does the following system of linear equations (with unknowns $x, y, z$)} \\
\text{have a unique solution?}
$$
\begin{cases}
x - y + 3z = 1 \\
2x + 7z = -3 \\
-3x + y + mz = 5
\end{cases}
$$
