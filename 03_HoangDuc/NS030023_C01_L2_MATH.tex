\section*{Problem}

\text{Use the definition of $\mathbf{A}\mathbf{x}$ to write the matrix equation as a vector equation, or vice versa:}

\subsection*{(a) Matrix Equation $\to$ Vector Equation}
$$
\begin{pmatrix}
7 & 2 & -9 & 3 \\
-4 & -5 & 7 & -2
\end{pmatrix}
\begin{pmatrix}
x_1 \\ x_2 \\ x_3 \\ x_4
\end{pmatrix}
=
\begin{pmatrix}
-9 \\ 44
\end{pmatrix}
$$

\subsection*{(b) Matrix Equation $\to$ Vector Equation}
$$
\begin{pmatrix}
7 & -3 \\
2 & 1 \\
9 & -6 \\
-3 & 2
\end{pmatrix}
\begin{pmatrix}
x_1 \\ x_2
\end{pmatrix}
=
\begin{pmatrix}
1 \\ -9 \\ 12 \\ -4
\end{pmatrix}
$$

\subsection*{(c) Vector Equation $\to$ Matrix Equation}
$$
x_1 \begin{pmatrix} 4 \\ -1 \\ 7 \\ -4 \end{pmatrix} + x_2 \begin{pmatrix} -5 \\ 3 \\ -5 \\ 1 \end{pmatrix} + x_3 \begin{pmatrix} 7 \\ -8 \\ 0 \\ 2 \end{pmatrix} = \begin{pmatrix} 6 \\ -8 \\ 0 \\ -7 \end{pmatrix}
$$

\subsection*{(d) Vector Equation $\to$ Matrix Equation}
$$
z_1 \begin{pmatrix} 4 \\ -2 \end{pmatrix} + z_2 \begin{pmatrix} -4 \\ 5 \end{pmatrix} + z_3 \begin{pmatrix} -5 \\ 4 \end{pmatrix} + z_4 \begin{pmatrix} 3 \\ 0 \end{pmatrix} = \begin{pmatrix} 4 \\ 13 \end{pmatrix}
$$
