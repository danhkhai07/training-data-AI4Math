\section*{Problem}

\text{Find $\mathbf{A} + \mathbf{B}$, $\mathbf{A} - \mathbf{B}$, $2\mathbf{A}$, and $\mathbf{B} + \frac{1}{2}\mathbf{A}$ for the given matrices $\mathbf{A}$ and $\mathbf{B}$:}

\subsection*{(a)}
$$
\mathbf{A} = \begin{pmatrix} 6 & -1 \\ 2 & 4 \\ -3 & 5 \end{pmatrix}, \quad \mathbf{B} = \begin{pmatrix} 1 & 4 \\ -1 & 5 \\ 1 & 10 \end{pmatrix}
$$

\subsection*{(b)}
$$
\mathbf{A} = \begin{pmatrix} 2 & 1 & 1 \\ -1 & -1 & 4 \end{pmatrix}, \quad \mathbf{B} = \begin{pmatrix} 2 & -3 & 4 \\ -3 & 1 & -2 \end{pmatrix}
$$
