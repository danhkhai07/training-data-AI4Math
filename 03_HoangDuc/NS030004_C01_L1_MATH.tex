\section*{Problem}

\text{Solve the following systems of equations:}

\subsection*{(a)}
\text{System of 2 equations with variables $1/x$ and $1/y$:}
$$
\begin{cases}
\frac{2}{x} + \frac{3}{y} = 0 \\
\frac{3}{x} - \frac{4}{y} = -\frac{25}{6}
\end{cases}
$$

\subsection*{(b)}
\text{System of 3 equations with variables $1/x$, $1/y$, and $1/z$:}
$$
\begin{cases}
\frac{2}{x} + \frac{1}{y} - \frac{3}{z} = 4 \\
\frac{4}{x} + \frac{2}{z} = 10 \\
-\frac{2}{x} + \frac{3}{y} - \frac{13}{z} = -8
\end{cases}
$$
