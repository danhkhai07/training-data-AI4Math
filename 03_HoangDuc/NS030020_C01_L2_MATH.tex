\section*{Problem}

\text{Which of the following sets span $\mathbf{R}^3$:}

\subsection*{(a)}
$$
\left\{ \begin{pmatrix} 4 \\ 7 \\ 3 \end{pmatrix}, \begin{pmatrix} -1 \\ 2 \\ 6 \end{pmatrix}, \begin{pmatrix} 2 \\ -3 \\ 5 \end{pmatrix} \right\}
$$

\subsection*{(b)}
$$
\left\{ \begin{pmatrix} 6 \\ 7 \\ 6 \end{pmatrix}, \begin{pmatrix} 3 \\ 2 \\ -4 \end{pmatrix}, \begin{pmatrix} 1 \\ -3 \\ 2 \end{pmatrix} \right\}
$$

\subsection*{(c)}
$$
\left\{ \begin{pmatrix} -2 \\ 5 \\ 0 \end{pmatrix}, \begin{pmatrix} 4 \\ 6 \\ 3 \end{pmatrix} \right\}
$$

\subsection*{(d)}
$$
\left\{ \begin{pmatrix} 1 \\ 1 \\ 0 \end{pmatrix}, \begin{pmatrix} 1 \\ 0 \\ 1 \end{pmatrix}, \begin{pmatrix} 0 \\ 1 \\ 1 \end{pmatrix} \right\}
$$

\subsection*{(e)}
$$
\left\{ \begin{pmatrix} 1 \\ -2 \\ 0 \end{pmatrix}, \begin{pmatrix} 0 \\ 0 \\ 1 \end{pmatrix}, \begin{pmatrix} -1 \\ 2 \\ 0 \end{pmatrix} \right\}
$$

\subsection*{(f)}
$$
\left\{ \begin{pmatrix} 1 \\ 0 \\ 3 \end{pmatrix}, \begin{pmatrix} 2 \\ 0 \\ -1 \end{pmatrix}, \begin{pmatrix} 4 \\ 0 \\ 5 \end{pmatrix}, \begin{pmatrix} 2 \\ 0 \\ 6 \end{pmatrix} \right\}
$$
