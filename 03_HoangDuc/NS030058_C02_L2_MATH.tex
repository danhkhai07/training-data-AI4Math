\section*{Problem}

\text{Given matrices $\mathbf{A}$ and $\mathbf{B}$ satisfy the following equations:}
$$
\mathbf{A} + 2\mathbf{B} = \begin{pmatrix} 2 & -3 \\ 1 & 5 \end{pmatrix}
$$
$$
2\mathbf{A} + 5\mathbf{B} = \begin{pmatrix} 7 & 5 \\ 2 & 3 \end{pmatrix}
$$
\text{Find the matrix $\mathbf{B}$.}
