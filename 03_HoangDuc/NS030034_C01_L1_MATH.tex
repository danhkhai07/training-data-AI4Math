\subsection*{(a)}
$$
\begin{cases}
x + 2y - 4z = 0 \\

\end{cases}
$$

\subsection*{(b)}
$$
\begin{cases}
x - 2y - 9z + 5t = 0 \\
x - y - 7z - t = 0
\end{cases}
$$
