\section*{Problem}

\text{Calculate the determinant of the following matrices by cofactor expansion along a row or column, and using elementary row/column operations (if necessary):}

\subsection*{(a)}
$$
\begin{pmatrix}
1 & 0 & 2 \\
-1 & 1 & 4 \\
2 & 0 & 3
\end{pmatrix}
$$

\subsection*{(b)}
$$
\begin{pmatrix}
1 & 4 & -2 \\
3 & 2 & 0 \\
-1 & 4 & 3
\end{pmatrix}
$$

\subsection*{(c)}
$$
\begin{pmatrix}
2 & -1 & 3 \\
1 & 4 & 4 \\
1 & 0 & 2
\end{pmatrix}
$$

\subsection*{(d)}
$$
\begin{pmatrix}
x & y & 1 \\
2 & 3 & 1 \\
0 & -1 & 1
\end{pmatrix}
$$

\subsection*{(e)}
$$
\begin{pmatrix}
x & y & 1 \\
-2 & -2 & 1 \\
1 & 5 & 1
\end{pmatrix}
$$

\subsection*{(f)}
$$
\begin{pmatrix}
2 & 6 & 6 & 2 \\
2 & 7 & 3 & 6 \\
1 & 5 & 0 & 1 \\
3 & 7 & 0 & 7
\end{pmatrix}
$$

\subsection*{(g)}
$$
\begin{pmatrix}
1 & 4 & 3 & 2 \\
-5 & 6 & 2 & 1 \\
0 & 0 & 0 & 0 \\
3 & -2 & 1 & 5
\end{pmatrix}
$$

\subsection*{(h)}
$$
\begin{pmatrix}
5 & 3 & 0 & 6 \\
4 & 6 & 4 & 12 \\
0 & 2 & -3 & 4 \\
0 & 1 & -2 & 2
\end{pmatrix}
$$

\subsection*{(i)}
$$
\begin{pmatrix}
3 & 0 & 7 & 0 \\
2 & 6 & 11 & 12 \\
4 & 1 & -1 & 2 \\
1 & 5 & 2 & 10
\end{pmatrix}
$$

\subsection*{(j)}
$$
\begin{pmatrix}
5 & 2 & 0 & 0 & -2 \\
0 & 1 & 4 & 3 & 2 \\
0 & 0 & 2 & 6 & 3 \\
0 & 0 & 3 & 4 & 1 \\
0 & 0 & 0 & 0 & 2
\end{pmatrix}
$$
