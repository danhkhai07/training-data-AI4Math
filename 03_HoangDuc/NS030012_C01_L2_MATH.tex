\section*{Problem}

\text{Determine if $\mathbf{b}$ is a linear combination of $\mathbf{a}_1$, $\mathbf{a}_2$, and $\mathbf{a}_3$, where:}

\subsection*{(a)}
$$
\mathbf{a}_1 = \begin{pmatrix} 1 \\ -2 \\ 0 \end{pmatrix}, \quad \mathbf{a}_2 = \begin{pmatrix} 0 \\ 1 \\ 2 \end{pmatrix}, \quad \mathbf{a}_3 = \begin{pmatrix} 5 \\ -6 \\ 8 \end{pmatrix}, \quad \mathbf{b} = \begin{pmatrix} 2 \\ -1 \\ 6 \end{pmatrix}
$$

\subsection*{(b)}
$$
\mathbf{a}_1 = \begin{pmatrix} 1 \\ -2 \\ 2 \end{pmatrix}, \quad \mathbf{a}_2 = \begin{pmatrix} 0 \\ 5 \\ 5 \end{pmatrix}, \quad \mathbf{a}_3 = \begin{pmatrix} 2 \\ 0 \\ 8 \end{pmatrix}, \quad \mathbf{b} = \begin{pmatrix} -5 \\ 11 \\ -7 \end{pmatrix}
$$
