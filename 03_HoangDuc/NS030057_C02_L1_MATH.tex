\section*{Problem}

\text{Find the matrix $\mathbf{X}$ given the matrices $\mathbf{A}$ and $\mathbf{B}$:}
$$
\mathbf{A} = \begin{pmatrix} -4 & 0 \\ 1 & -5 \\ -3 & 2 \end{pmatrix}, \quad \mathbf{B} = \begin{pmatrix} 1 & 2 \\ -2 & 1 \\ 4 & 4 \end{pmatrix}
$$

\subsection*{(a)}
$$
3\mathbf{X} + 2\mathbf{A} = \mathbf{B}
$$

\subsection*{(b)}
$$
2\mathbf{A} - 5\mathbf{B} = 3\mathbf{X}
$$

\subsection*{(c)}
$$
\mathbf{X} - 3\mathbf{A} + 2\mathbf{B} = \mathbf{0}
$$

\subsection*{(d)}
$$
6\mathbf{X} - 4\mathbf{A} - 3\mathbf{B} = \mathbf{0}
$$
