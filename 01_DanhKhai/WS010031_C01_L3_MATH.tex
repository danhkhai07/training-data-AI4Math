\subsection*{Cho hệ phương trình}
$$
\begin{cases}
x + 2y + az = 3 \\
3x - y - az = 2 \\
2x + y + 3z = b
\end{cases}
$$
\begin{enumerate}
    \item[a)] Tìm $a, b$ để hệ có nghiệm duy nhất.
    \item[b)] Tìm $a, b$ để hệ có vô số nghiệm.
\end{enumerate}

\section*{Giải.}

\text{Ta xét ma trận hệ số mở rộng $\bar{A}$:}
$$
\bar{A} = \begin{pmatrix}
1 & 2 & a & | & 3 \\
3 & -1 & -a & | & 2 \\
2 & 1 & 3 & | & b
\end{pmatrix}
$$
\text{Thực hiện phép biến đổi $\text{H}_2 \leftarrow \text{H}_2 - 3\text{H}_1$ và $\text{H}_3 \leftarrow \text{H}_3 - 2\text{H}_1$:}
$$
\xrightarrow[\text{H}_3 \leftarrow \text{H}_3 - 2\text{H}_1]{\text{H}_2 \leftarrow \text{H}_2 - 3\text{H}_1}
\begin{pmatrix}
1 & 2 & a & | & 3 \\
0 & -7 & -4a & | & -7 \\
0 & -3 & 3-2a & | & b-6
\end{pmatrix}
$$
\text{Thực hiện $\text{H}_2 \leftarrow -\frac{1}{7}\text{H}_2$:}
$$
\xrightarrow{\text{H}_2 \leftarrow -\frac{1}{7}\text{H}_2}
\begin{pmatrix}
1 & 2 & a & | & 3 \\
0 & 1 & \frac{4a}{7} & | & 1 \\
0 & -3 & 3-2a & | & b-6
\end{pmatrix}
$$
\text{Thực hiện $\text{H}_3 \leftarrow \text{H}_3 + 3\text{H}_2$:}
$$
\xrightarrow{\text{H}_3 \leftarrow \text{H}_3 + 3\text{H}_2}
\begin{pmatrix}
1 & 2 & a & | & 3 \\
0 & 1 & \frac{4a}{7} & | & 1 \\
0 & 0 & \frac{21 - 2a}{7} & | & b - 3
\end{pmatrix}
$$

\subsection*{a) Hệ có nghiệm duy nhất}
\text{Hệ có nghiệm duy nhất khi và chỉ khi $r(A) = r(\bar{A}) = 3$. }
\text{Điều này xảy ra khi phần tử dẫn đầu hàng 3 khác 0:}
$$
\frac{21 - 2a}{7} \ne 0 \Leftrightarrow a \ne \frac{21}{2}.
$$
\text{Giá trị $b$ có thể là bất kỳ số thực nào.}
\text{Vậy hệ có nghiệm duy nhất khi $a \ne \frac{21}{2}$ và $b \in \mathbb{R}$.}

\subsection*{b) Hệ có vô số nghiệm}
\text{Hệ có vô số nghiệm khi và chỉ khi $r(A) = r(\bar{A}) < 3$. }
\text{Điều này xảy ra khi hàng 3 là hàng không, tức là:}
\begin{cases}
\frac{21 - 2a}{7} = 0 \\
b - 3 = 0
\end{cases}
$$
\text{Tức là $a = \frac{21}{2}$ và $b = 3$.}
