\subsection*{Giải hệ phương trình sau}
$$
\begin{cases}
2x_1 + 5x_2 - 8x_3 = 8 \\
4x_1 + 3x_2 - 9x_3 = 9 \\
2x_1 + 3x_2 - 5x_3 = 7 \\
x_1 + 8x_2 - 7x_3 = 12
\end{cases}
$$

\section*{Giải.}

\text{Sử dụng phương pháp Gauss, biến đổi ma trận hệ số mở rộng $\bar{A}$:}
$$
\bar{A} = \begin{pmatrix}
2 & 5 & -8 & | & 8 \\
4 & 3 & -9 & | & 9 \\
2 & 3 & -5 & | & 7 \\
1 & 8 & -7 & | & 12
\end{pmatrix}
$$
\text{Thực hiện $\text{H}_1 \leftrightarrow \text{H}_4$ (để có phần tử dẫn đầu bằng 1):}
$$
\xrightarrow{\text{H}_1 \leftrightarrow \text{H}_4}
\begin{pmatrix}
1 & 8 & -7 & | & 12 \\
4 & 3 & -9 & | & 9 \\
2 & 3 & -5 & | & 7 \\
2 & 5 & -8 & | & 8
\end{pmatrix}
$$
\text{Thực hiện $\text{H}_2 \leftarrow \text{H}_2 - 4\text{H}_1$, $\text{H}_3 \leftarrow \text{H}_3 - 2\text{H}_1$, $\text{H}_4 \leftarrow \text{H}_4 - 2\text{H}_1$:}
$$
\xrightarrow[\text{H}_4 \leftarrow \text{H}_4 - 2\text{H}_1]{\text{H}_2 \leftarrow \text{H}_2 - 4\text{H}_1, \text{H}_3 \leftarrow \text{H}_3 - 2\text{H}_1}
\begin{pmatrix}
1 & 8 & -7 & | & 12 \\
0 & -29 & 19 & | & -39 \\
0 & -13 & 9 & | & -17 \\
0 & -11 & 6 & | & -16
\end{pmatrix}
$$
\text{Thực hiện $\text{H}_2 \leftrightarrow \text{H}_4$ (để có số nhỏ hơn làm phần tử dẫn đầu):}
$$
\xrightarrow{\text{H}_2 \leftrightarrow \text{H}_4}
\begin{pmatrix}
1 & 8 & -7 & | & 12 \\
0 & -11 & 6 & | & -16 \\
0 & -13 & 9 & | & -17 \\
0 & -29 & 19 & | & -39
\end{pmatrix}
$$
\text{Thực hiện $\text{H}_3 \leftarrow 11\text{H}_3 - 13\text{H}_2$ và $\text{H}_4 \leftarrow 11\text{H}_4 - 29\text{H}_2$:}
$$
\xrightarrow[\text{H}_4 \leftarrow 11\text{H}_4 - 29\text{H}_2]{\text{H}_3 \leftarrow 11\text{H}_3 - 13\text{H}_2}
\begin{pmatrix}
1 & 8 & -7 & | & 12 \\
0 & -11 & 6 & | & -16 \\
0 & 0 & 21 & | & 21 \\
0 & 0 & 35 & | & 35
\end{pmatrix}
$$
\text{Thực hiện $\text{H}_3 \leftarrow \frac{1}{21}\text{H}_3$ và $\text{H}_4 \leftarrow \text{H}_4 - \frac{35}{21}\text{H}_3$ (hay $\text{H}_4 \leftarrow \text{H}_4 - \frac{5}{3}\text{H}_3$):}
$$
\xrightarrow[\text{H}_4 \leftarrow \text{H}_4 - \frac{5}{3}\text{H}_3]{\text{H}_3 \leftarrow \frac{1}{21}\text{H}_3}
\begin{pmatrix}
1 & 8 & -7 & | & 12 \\
0 & -11 & 6 & | & -16 \\
0 & 0 & 1 & | & 1 \\
0 & 0 & 0 & | & 0
\end{pmatrix}
$$
\text{Ma trận đã đạt dạng bậc thang. Hạng $r(A) = r(\bar{A}) = 3$. Hệ có nghiệm duy nhất ($3$ ẩn).}

\text{Hệ phương trình tương đương là:}
$$
\begin{cases}
x_1 + 8x_2 - 7x_3 = 12 \\
-11x_2 + 6x_3 = -16 \\
x_3 = 1
\end{cases}
$$
\text{Giải ngược từ dưới lên:}
\begin{itemize}
    \item $x_3 = 1$.
    \item Thay $x_3 = 1$ vào phương trình thứ hai: $-11x_2 + 6(1) = -16 \Rightarrow -11x_2 = -22 \Rightarrow x_2 = 2$.
    \item Thay $x_2 = 2, x_3 = 1$ vào phương trình thứ nhất: $x_1 + 8(2) - 7(1) = 12 \Rightarrow x_1 + 16 - 7 = 12 \Rightarrow x_1 + 9 = 12 \Rightarrow x_1 = 3$.
\end{itemize}

\text{Vậy nghiệm của hệ phương trình là $(x_1, x_2, x_3) = (3, 2, 1)$.}
