\subsection*{Tìm ma trận $X$ thỏa mãn phương trình}
\begin{enumerate}
    \item[a)] $\begin{pmatrix} 2 & 1 \\ 5 & 3 \end{pmatrix} X = \begin{pmatrix} 1 & 1 \\ 0 & 2 \end{pmatrix}$.
    \item[b)] $\begin{pmatrix} 2 & 3 & 1 \\ -1 & 0 & 1 \\ -1 & 2 & 1 \end{pmatrix} X = \begin{pmatrix} -1 & 0 \\ 1 & 1 \\ 0 & 1 \end{pmatrix}$.
\end{enumerate}

\section*{Giải.}

\text{Ta có phương trình ma trận dạng $AX = B$. Nếu $A$ khả nghịch, thì $X = A^{-1}B$. Ta sử dụng phương pháp ma trận mở rộng $(A|B)$ để tìm $X$.}

\subsection*{a) Ta xét ma trận mở rộng $(A|B)$:}
$$
\begin{pmatrix}
2 & 1 & | & 1 & 1 \\
5 & 3 & | & 0 & 2
\end{pmatrix}
$$
\text{Thực hiện phép biến đổi $\text{H}_1 \leftarrow \text{H}_1 - \text{H}_2$:}
$$
\xrightarrow{\text{H}_1 \leftarrow \text{H}_1 - \text{H}_2}
\begin{pmatrix}
-3 & -2 & | & 1 & -1 \\
5 & 3 & | & 0 & 2
\end{pmatrix}
$$
\text{Thực hiện $\text{H}_2 \leftarrow \text{H}_2 + \text{H}_1$:}
$$
\xrightarrow{\text{H}_2 \leftarrow \text{H}_2 + \text{H}_1}
\begin{pmatrix}
-3 & -2 & | & 1 & -1 \\
2 & 1 & | & 1 & 1
\end{pmatrix}
$$
\text{Thực hiện $\text{H}_1 \leftarrow \text{H}_1 + 2\text{H}_2$:}
$$
\xrightarrow{\text{H}_1 \leftarrow \text{H}_1 + 2\text{H}_2}
\begin{pmatrix}
1 & 0 & | & 3 & 1 \\
2 & 1 & | & 1 & 1
\end{pmatrix}
$$
\text{Thực hiện $\text{H}_2 \leftarrow \text{H}_2 - 2\text{H}_1$:}
$$
\xrightarrow{\text{H}_2 \leftarrow \text{H}_2 - 2\text{H}_1}
\begin{pmatrix}
1 & 0 & | & 3 & 1 \\
0 & 1 & | & -5 & -1
\end{pmatrix}
$$
\text{Vậy}
$$
X = \begin{pmatrix} 3 & 1 \\ -5 & -1 \end{pmatrix}.
$$

\subsection*{b) Ta xét ma trận mở rộng $(A|B)$:}
$$
\begin{pmatrix}
2 & 3 & 1 & | & -1 & 0 \\
-1 & 0 & 1 & | & 1 & 1 \\
-1 & 2 & 1 & | & 0 & 1
\end{pmatrix}
$$
\text{Thực hiện $\text{H}_1 \leftrightarrow \text{H}_2$ (và $\text{H}_2 \leftarrow -\text{H}_2$ để làm gọn):}
$$
\xrightarrow{\text{H}_1 \leftrightarrow \text{H}_2}
\begin{pmatrix}
-1 & 0 & 1 & | & 1 & 1 \\
2 & 3 & 1 & | & -1 & 0 \\
-1 & 2 & 1 & | & 0 & 1
\end{pmatrix}
\xrightarrow{\text{H}_1 \leftarrow -\text{H}_1}
\begin{pmatrix}
1 & 0 & -1 & | & -1 & -1 \\
2 & 3 & 1 & | & -1 & 0 \\
-1 & 2 & 1 & | & 0 & 1
\end{pmatrix}
$$
\text{Thực hiện $\text{H}_2 \leftarrow \text{H}_2 - 2\text{H}_1$, $\text{H}_3 \leftarrow \text{H}_3 + \text{H}_1$:}
$$
\xrightarrow[\text{H}_3 \leftarrow \text{H}_3 + \text{H}_1]{\text{H}_2 \leftarrow \text{H}_2 - 2\text{H}_1}
\begin{pmatrix}
1 & 0 & -1 & | & -1 & -1 \\
0 & 3 & 3 & | & 1 & 2 \\
0 & 2 & 0 & | & -1 & 0
\end{pmatrix}
$$
\text{Thực hiện $\text{H}_3 \leftarrow \frac{1}{2}\text{H}_3$ (để có $a_{32}=1$):}
$$
\xrightarrow{\text{H}_3 \leftarrow \frac{1}{2}\text{H}_3}
\begin{pmatrix}
1 & 0 & -1 & | & -1 & -1 \\
0 & 3 & 3 & | & 1 & 2 \\
0 & 1 & 0 & | & -\frac{1}{2} & 0
\end{pmatrix}
$$
\text{Thực hiện $\text{H}_2 \leftarrow \text{H}_2 - 3\text{H}_3$:}
$$
\xrightarrow{\text{H}_2 \leftarrow \text{H}_2 - 3\text{H}_3}
\begin{pmatrix}
1 & 0 & -1 & | & -1 & -1 \\
0 & 0 & 3 & | & \frac{5}{2} & 2 \\
0 & 1 & 0 & | & -\frac{1}{2} & 0
\end{pmatrix}
$$
\text{Đổi vị trí $\text{H}_2 \leftrightarrow \text{H}_3$ và $\text{H}_2 \leftarrow \frac{1}{3}\text{H}_2$:}
$$
\xrightarrow{\text{H}_2 \leftrightarrow \text{H}_3}
\begin{pmatrix}
1 & 0 & -1 & | & -1 & -1 \\
0 & 1 & 0 & | & -\frac{1}{2} & 0 \\
0 & 0 & 3 & | & \frac{5}{2} & 2
\end{pmatrix}
\xrightarrow{\text{H}_3 \leftarrow \frac{1}{3}\text{H}_3}
\begin{pmatrix}
1 & 0 & -1 & | & -1 & -1 \\
0 & 1 & 0 & | & -\frac{1}{2} & 0 \\
0 & 0 & 1 & | & \frac{5}{6} & \frac{2}{3}
\end{pmatrix}
$$
\text{Thực hiện $\text{H}_1 \leftarrow \text{H}_1 + \text{H}_3$:}
$$
\xrightarrow{\text{H}_1 \leftarrow \text{H}_1 + \text{H}_3}
\begin{pmatrix}
1 & 0 & 0 & | & -\frac{1}{6} & -\frac{1}{3} \\
0 & 1 & 0 & | & -\frac{1}{2} & 0 \\
0 & 0 & 1 & | & \frac{5}{6} & \frac{2}{3}
\end{pmatrix}
$$
\text{Vậy}
$$
X = \begin{pmatrix}
-\frac{1}{6} & -\frac{1}{3} \\
-\frac{1}{2} & 0 \\
\frac{5}{6} & \frac{2}{3}
\end{pmatrix} = \frac{1}{6} \begin{pmatrix}
-1 & -2 \\
-3 & 0 \\
5 & 4
\end{pmatrix}.
$$
