\subsection*{Bài 2.1.26. Cho các véctơ trong $\mathbb{R}^4$}
$$
\mathbf{a} = (1, -2, 3, 2), \mathbf{b} = (2, 0, -1, 4), \mathbf{c} = (3, -2, -1, 3), \mathbf{d} = (3, -6, 6, 3).
$$
\text{Chứng minh rằng } $\mathcal{L}(\mathbf{a}, \mathbf{b}, \mathbf{c}, \mathbf{d}) = \mathcal{L}(\mathbf{a}, \mathbf{b}, \mathbf{c})$. \text{Tìm hạng của hệ các véctơ } $\{\mathbf{a}, \mathbf{b}, \mathbf{c}, \mathbf{d}\}$.

\section*{Giải.}

\text{Trước hết, ta đi tìm hạng của hệ véctơ } $\{\mathbf{a}, \mathbf{b}, \mathbf{c}, \mathbf{d}\}$. \text{Xét ma trận tọa độ của các véctơ } $\{\mathbf{a}, \mathbf{b}, \mathbf{c}, \mathbf{d}\}$ \text{ trong cơ sở chính tắc của } $\mathbb{R}^4$:
$$
A = \begin{pmatrix}
1 & 2 & 3 & 3 \\
-2 & 0 & -2 & -6 \\
3 & -1 & -1 & 6 \\
2 & 4 & 3 & 3
\end{pmatrix}
$$
\text{Biến đổi sơ cấp về hàng của ma trận } $A$:
$$
A \xrightarrow{H_2 \to 2H_1+H_2}
\xrightarrow{H_3 \to -3H_1+H_3}
\xrightarrow{H_4 \to -2H_1+H_4}
\begin{pmatrix}
1 & 2 & 3 & 3 \\
0 & 4 & 4 & 0 \\
0 & -7 & -10 & -3 \\
0 & 0 & -3 & -3
\end{pmatrix}
$$
$$
\xrightarrow{H_3 \to \frac{7}{4}H_2+H_3}
\begin{pmatrix}
1 & 2 & 3 & 3 \\
0 & 4 & 4 & 0 \\
0 & 0 & -3 & -3 \\
0 & 0 & -3 & -3
\end{pmatrix}
$$
$$
\xrightarrow{H_4 \to -H_3+H_4}
\begin{pmatrix}
1 & 2 & 3 & 3 \\
0 & 4 & 4 & 0 \\
0 & 0 & -3 & -3 \\
0 & 0 & 0 & 0
\end{pmatrix}
$$
\text{Ta thu được } $r(A) = 3$. \text{Do đó } $r(\{\mathbf{a}, \mathbf{b}, \mathbf{c}, \mathbf{d}\}) = r(A) = 3$.

\text{Từ ma trận cuối cùng trong biến đổi sơ cấp về hàng của ma trận } $A$, \text{ta suy ra } $r(\{\mathbf{a}, \mathbf{b}, \mathbf{c}\}) = r(\{\mathbf{a}, \mathbf{b}, \mathbf{c}, \mathbf{d}\}) = 3$. \text{Điều đó chứng tỏ } $\mathcal{L}(\mathbf{a}, \mathbf{b}, \mathbf{c}, \mathbf{d}) = \mathcal{L}(\mathbf{a}, \mathbf{b}, \mathbf{c})$.
